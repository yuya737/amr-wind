139  amrex::Vector<amrex::MultiFab*> get_mfab_vec(int lev)
140  {
141  const int nstates = amrex::min(m_field.num_time_states(), 2);
142  amrex::Vector<amrex::MultiFab*> ret;
143 
144  // The states in the FieldInfo data are ordered from newest to oldest,
145  // so swap the order
146  for (int i = nstates - 1; i >= 0; --i) {
147  const auto fstate = static_cast<FieldState>(i);
148  ret.push_back(&m_field.state(fstate)(lev));
149  }
150  return ret;
151  }
